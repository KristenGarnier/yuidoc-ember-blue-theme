import moment from 'moment'
import {constructRequest as requester, base64Encode as encoder, makeItSpin, logPerformance} from '../utils'
import config  from '../config'
import human from 'humane-js'
import i18n from 'i18next'
import '../../node_modules/humane-js/themes/flatty.css'
import './human.css'
import * as actions from '../model/actions/actions'


/**
 * Gestion des callback pour les listeners du DOM
 *
 * @module domListenerCallback
 * @class domListenerCallback
 */


/**
 * Permet de changer le state d'une checkbox quand on clique dessus
 *
 * Utilisé comme callback sur l'évènement 'change', sur les checkboxs.
 * Déclanche l'action `checkedSearch dans le model`
 *
 * @method handleCheck
 * @param {Object} store Instance de du store de redux
 * @param {Object} event Evènement renvoyé par l'écouteur d'évènement
 */
export const handleCheck = (store, event) => {
    store.dispatch(actions.chekedSearch({id: event.target.name, state: event.target.checked}))
}

/**
 * Permet de mettre en forme la query et de réinstancer un nouveau state
 *
 * Enregistrement des paramètres de query dans le localhost, et enregistrement du timestamp pour l'invalidation
 *
 * @method handleEngineUpdate
 * @param {object} store redux object
 * @param {object} event
 *
 * @since 0.2.0
 */
export const handleEngineUpdate = (store, event) => {
    event.preventDefault()
    event.stopPropagation()

    if (moment(event.target.querySelector(`input[name="${config.form.start}"]`).value, 'D/M/Y').isValid() && moment(event.target.querySelector(`input[name="${config.form.end}"]`).value, 'D/M/Y').isValid()) {

        const query = {
            start: moment(event.target.querySelector(`input[name="${config.form.start}"]`).value, 'D/M/Y').format('Y/M/D'),
            end: moment(event.target.querySelector(`input[name="${config.form.end}"]`).value, 'D/M/Y').format('Y/M/D'),
            type: event.target.querySelector(`select[name="${config.form.type}"]`).value,
            person: event.target.querySelector(`select[name="${config.form.person}"]`).value
        }

        localStorage.setItem('dates', JSON.stringify(query))
        localStorage.setItem('dates:timestamp', JSON.stringify(moment.utc(moment.now())))

        if(!config.form.hasOwnProperty('ajax') || config.form.ajax) {
            const timeStartRequest = moment()
            requester(config.baseUrl, config.endpoint, encoder(query))
                .then((res) => {
                    makeItSpin(false)
                    logPerformance(
                        'results-load-time',
                        config.ip,
                        moment().diff(timeStartRequest, 'seconds'),
                    )
                    store.dispatch(actions.initiateState(res.data))
                })

            if (window.hasOwnProperty('eventManager')) {
                window.eventManager.emit('UPDATE_DATES', Object.assign({}, query, {
                    room_type: config.room_type
                }))
            }

            if(window.hasOwnProperty('STRAT_AIRBNB_CONFIG') && document.querySelector('.' + window.STRAT_AIRBNB_CONFIG.renders.airbnbCalendar)) {
                document.querySelector('.' + window.STRAT_AIRBNB_CONFIG.renders.airbnbCalendar + ' button').click()
            }
        } else {
            event.target.submit()
        }

    } else {
        human.log(i18n.t('error.form'), {timeout: 4000, clickToClose: true, addnCls: 'human-error'})
    }
}


export default {
    handleCheck,
    handleEngineUpdate
}
