function process() {
    /**
     * En charge de lancer les rendu sur les différentes parties de l'app
     *
     * Déclenche un rendu dans les différentes parties de l'app, quand une donnée du model est modifiée
     *
     * @method render
     * @param {Object} store Instance de la classe AjaxSearch
     */
    const render = (store) => {

        if (window.CONFIG.resalysList && document.getElementById(window.CONFIG.resalysList)) {
            const proposals = proposalData(store.getState().proposals)
            const data = {
                props: filterOutCounter(proposals),
                counterProps: filterOutNotCounter(proposals),
                hasCounter: hasProposalsCounter(proposals),
                date: JSON.parse(localStorage.getItem('dates')) || {
                    start: moment(window.CONFIG.form.default_start),
                    end: moment(window.CONFIG.form.default_end)
                }
            }

            if (data.props.length === 0 && data.counterProps.length === 0) {
                human.log(i18n.t('error.nodata'), {timeout: 4000, clickToClose: true, addnCls: 'human-error'})

                logNoResults('noresult', searchFormData(store.getState().proposals))

                store.dispatch(actions.chekedSearch({
                    id: store.getState().history.data.id,
                    state: !store.getState().history.data.state
                }))

            } else {
                renderer(store, data, 'proposals', postTemplate, config.resalysList, null)
            }

            if (window.hasOwnProperty('eventManager')) {
                const nbResults = proposals.reduce((previous, current) => {
                    return current.proposals.length + previous
                }, 0)
                window.eventManager.emit('LOG_RESULT', nbResults)
            }

        }
        if (window.CONFIG.facet && document.getElementById(window.CONFIG.facet)) {
            if(proposalData(store.getState().proposals)[0].term_id === undefined) {
                renderer(store, searchFormData(store.getState().proposals), 'search', searchFormTemplate, config.facet, curryCb(handleCheck), true)
                document.querySelector('main .j-col-9').className = 'j-wrapper-full'
            } else {
                renderer(store, searchFormData(store.getState().proposals), 'search', searchFormTemplate, config.facet, curryCb(handleCheck))
            }
        }

        if (window.hasOwnProperty('eventManager')) {
            window.eventManager.emit('RENDER_FINISHED')
        }
    }

    /**
     * En charge de la gestion de l'historique sur le navigateur
     *
     * @method historyChanger
     * @param {Object} history Instance de la classe AjaxSearch
     */
    const historyChanger = (history) => {
        //browserHistory.push('/test/test2', {hello: 'wolrd'})
        //browserHistory.goBack()
    }

    /**
     * Initiaisation de l'internationalisation
     */
    i18n.init(i18nConfig, (err) => {

        if (!err) {
            /**
             * Instance de l'ajax store
             *
             * @type {AjaxStore}
             */
                //const ajaxStore = new AjaxStore([render, historyChanger])
            const ajaxStore = Store()
            ajaxStore.subscribe(() => {
                render(ajaxStore)
            })

            /**
             * Récupération initale des paramètres sur le serveur
             */

            if (document.getElementById(window.CONFIG.facet) || document.getElementById(window.CONFIG.resalysList)) {
                const timeStartRequest = moment()

                if(window.hasOwnProperty('QUERY_PRESEARCH')) {

                    constructRequest(config.baseUrl, config.endpoint, base64Encode(window.QUERY_PRESEARCH))
                        .then((res) => {
                            makeItSpin(false)
                            logPerformance(
                                'results-load-time',
                                config.ip,
                                moment().diff(timeStartRequest, 'seconds'),
                            )
                            ajaxStore.dispatch(actions.initiateState(res.data))
                        })

                    if (window.hasOwnProperty('eventManager')) {
                        window.eventManager.emit('DATE_PICKER_UPDATE', {
                            startDate: moment(window.QUERY_PRESEARCH.start, 'Y/MM/DD'),
                            endDate: moment(window.QUERY_PRESEARCH.end, 'Y/MM/DD')
                        })

                        document.querySelector(`select[name="${config.form.type}"]`).value = window.QUERY_PRESEARCH.type
                        document.querySelector(`select[name="${config.form.person}"]`).value = window.QUERY_PRESEARCH.person
                    }

                } else if (localStorage.getItem('dates') && isTimeSmallerThan(JSON.parse(localStorage.getItem('dates:timestamp')), config.limit)) {
                    const dates = JSON.parse(localStorage.getItem('dates'))

                    constructRequest(config.baseUrl, config.endpoint, base64Encode(dates))
                        .then((res) => {
                            makeItSpin(false)
                            logPerformance(
                                'results-load-time',
                                config.ip,
                                moment().diff(timeStartRequest, 'seconds'),
                            )
                            ajaxStore.dispatch(actions.initiateState(res.data))
                        })
                } else {
                    constructRequest(config.baseUrl, config.endpoint)
                        .then((res) => {
                            makeItSpin(false)
                            logPerformance(
                                'results-load-time',
                                config.ip,
                                moment().diff(timeStartRequest, 'seconds'),

                            )
                            ajaxStore.dispatch(actions.initiateState(res.data))
                        })
                }

            }

            if (window.CONFIG.hasOwnProperty('engine') && window.CONFIG.hasOwnProperty('form') && document.getElementById(window.CONFIG.engine)) {
                /**
                 * Récupération des données sur le search engine
                 */
                document.querySelector(`#${config.engine}`).addEventListener('submit', curryCb(handleEngineUpdate)(ajaxStore))
            }
        } else {
            throw new Error(err)
        }

    })
}
