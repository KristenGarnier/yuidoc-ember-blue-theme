/**
 * Helper pour Handlebars permettant de faire des conditions ternaires
 *
 * Attention les helbers handlebars doivent respecter une configuration très spéciale.
 * En effet, si la configuration n'est pas respectée le helper ne sera pas reconnu et l'application ne pourrra pas fonctionner correctement.
 * La configuration est :
 * - Synthaxe ES5
 * - Utilisation du CommonJS
 * - Le nom du fichier doit être le nom de l'helper
 *
 * @module Template
 * @class Ternary
 *
 * @since 0.1.0
 */

module.exports = function (test, valueTrue, valueFalse) {
    return test ? valueTrue : valueFalse
}
